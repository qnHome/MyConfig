# zsh
```
# 更新
sudo apt update
# 安装
sudo apt install zsh
# 查看版本
zsh --version
# 设置默认shell
chsh -s /bin/zsh
# 集成 oh-my-zsh
wget https://github.com/robbyrussell/oh-my-zsh/raw/master/tools/install.sh -O - | sh
reboot
```
